## AI Agent Tool Contracts

### FileTools (Epic 12-1)

```typescript
// Used by AI to manipulate files
export const agentFileTools = {
  read_file: toolDefinition({
    name: 'read_file',
    description: 'Read contents of a file',
    parameters: z.object({
      path: z.string().describe('Relative path to file')
    }),
    execute: async ({ path }) => {
      return await syncManager.readFile(path);
    }
  }),
  
  write_file: toolDefinition({
    name: 'write_file',
    description: 'Write content to a file (creates if missing)',
    parameters: z.object({
      path: z.string(),
      content: z.string()
    }),
    execute: async ({ path, content }) => {
      await syncManager.writeFile(path, content);
      eventBus.emit('file:modified', { path, source: 'agent' });
    }
  }),
  
  list_directory: toolDefinition({
    name: 'list_directory',
    description: 'List files and folders in a directory',
    parameters: z.object({
      path: z.string().default('.')
    }),
    execute: async ({ path }) => {
      return await localFsAdapter.listDirectory(path);
    }
  })
};
```

### TerminalTools (Epic 12-2)

```typescript
// Used by AI to run commands
export const agentTerminalTools = {
  run_command: toolDefinition({
    name: 'run_command',
    description: 'Run a shell command and return output',
    parameters: z.object({
      command: z.string().describe('Command to execute')
    }),
    execute: async ({ command }) => {
      const [cmd, ...args] = command.split(' ');
      const process = await webContainer.spawn(cmd, args);
      const output = await process.output;
      return { stdout: output, exitCode: process.exit };
    }
  })
};
```
